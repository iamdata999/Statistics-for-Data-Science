import numpy as np
from scipy import stats
import pandas as pd

# Set a random seed for reproducibility
np.random.seed(0)

# Generate synthetic data for glucose levels (in mg/dL)
glucose_levels = np.random.normal(120, 20, 100)

# Generate synthetic data for BMI
bmi_values = np.random.normal(30, 5, 100)

# Create a synthetic diabetes dataset with Pandas
diabetes_data = pd.DataFrame({'Glucose Level': glucose_levels, 'BMI': bmi_values})

# Function to calculate confidence interval
def calculate_confidence_interval(data, confidence_level):
    # Calculate the sample mean and standard error
    sample_mean = np.mean(data)
    sample_std = np.std(data, ddof=1)  # Use ddof=1 for sample standard deviation
    sample_size = len(data)
    margin_of_error = 0

    # Calculate the critical value based on the confidence level
    alpha = 1 - confidence_level
    z_critical = stats.norm.ppf(1 - alpha / 2)

    # Calculate the margin of error
    margin_of_error = z_critical * (sample_std / np.sqrt(sample_size))

    # Calculate the confidence interval
    lower_limit = sample_mean - margin_of_error
    upper_limit = sample_mean + margin_of_error

    return lower_limit, upper_limit

# Input confidence level (0.75 for 75% confidence)
confidence_level = float(input("Enter the confidence level (e.g., 0.75 for 75% confidence): "))

# Calculate the confidence interval for glucose levels
glucose_lower, glucose_upper = calculate_confidence_interval(diabetes_data['Glucose Level'], confidence_level)

# Calculate the confidence interval for BMI values
bmi_lower, bmi_upper = calculate_confidence_interval(diabetes_data['BMI'], confidence_level)

# Print the results
print(f"Confidence Interval ({int(confidence_level * 100)}%):")
print(f"Glucose Level - Lower Limit: {glucose_lower}")
print(f"Glucose Level - Upper Limit: {glucose_upper}")
print(f"BMI - Lower Limit: {bmi_lower}")
print(f"BMI - Upper Limit: {bmi_upper}")

import random

def random_sample(population, sample_size):
    """
    Simulate random sampling from a population.

    Args:
        population (list or set): The population data to sample from.
        sample_size (int): The size of the desired sample.

    Returns:
        list: A random sample of size 'sample_size' from the population.
    """
    if sample_size > len(population):
        raise ValueError("Sample size cannot be greater than the population size.")

    # Use random.sample to obtain a random sample without replacement
    sample = random.sample(population, sample_size)
    return sample

# Example usage:

import numpy as np
import random

# Set a random seed for reproducibility
np.random.seed(0)

# Generate a synthetic dataset (e.g., normal distribution with mean=50, std_dev=10)
population = np.random.normal(loc=50, scale=10, size=1000)

# Specify the sample size
sample_size = 100

# Define the number of strata for stratified sampling
num_strata = 5

# Define the size of each stratum
stratum_size = sample_size // num_strata

# Define the sampling fraction for systematic sampling
systematic_sampling_fraction = sample_size / len(population)

# Define a function to calculate mean and variance
def calculate_mean_and_variance(data):
    mean = np.mean(data)
    variance = np.var(data, ddof=1)  # Use ddof=1 for sample variance
    return mean, variance

# Simple Random Sampling
simple_random_sample = random.sample(list(population), sample_size)
simple_random_mean, simple_random_variance = calculate_mean_and_variance(simple_random_sample)

# Stratified Sampling
strata = [random.sample(list(population[i:i + stratum_size]), stratum_size) for i in range(0, len(population), stratum_size)]
stratified_sample = [sample for stratum in strata for sample in stratum]
stratified_mean, stratified_variance = calculate_mean_and_variance(stratified_sample)

# Systematic Sampling
systematic_sample = [population[i] for i in range(0, len(population), int(1 / systematic_sampling_fraction))]
systematic_mean, systematic_variance = calculate_mean_and_variance(systematic_sample)

# Compare sample means and variances

print(f"Systematic Sampling: {systematic_mean}")
